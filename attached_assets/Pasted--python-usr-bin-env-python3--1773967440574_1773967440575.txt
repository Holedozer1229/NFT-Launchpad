# ------------------------------------------------------------------------------
# PRIME NUMBER GENERATION (Miller–Rabin)
# ------------------------------------------------------------------------------
def is_prime_mr(n: int, k: int = 5) -> bool:
    if n < 2:
        return False
    if n in (2, 3):
        return True
    if n % 2 == 0:
        return False
    d = n - 1
    s = 0
    while d % 2 == 0:
        d //= 2
        s += 1
    for _ in range(k):
        a = random.randint(2, n - 2)
        x = pow(a, d, n)
        if x == 1 or x == n - 1:
            continue
        for _ in range(s - 1):
            x = pow(x, 2, n)
            if x == n - 1:
                break
        else:
            return False
    return True

def generate_primes(n: int) -> List[int]:
    primes = []
    num = 2
    while len(primes) < n:
        if is_prime_mr(num):
            primes.append(num)
        num += 1
    return primes

# RIEMANN ZETA ZEROS (approximate)
# ------------------------------------------------------------------------------
def zeta_zero_approx(n: int) -> float:
    if n == 1:
        return 14.134725141734693790
    if n == 2:
        return 21.022039638771554993
    if n == 3:
        return 25.010857580145688763
    term = (n - 11/8) / math.e
    w = math.log(term) - math.log(math.log(term))
    return 2 * math.pi * (n - 11/8) / w

def spectral_gaps(n_zeros: int = 28) -> List[float]:
    zeros = [zeta_zero_approx(i+1) for i in range(n_zeros+1)]
    return [zeros[i+1] - zeros[i] for i in range(n_zeros)]

@dataclass
class Quaternion:
    a: float; b: float; c: float; d: float
    def __add__(self, other): return Quaternion(self.a+other.a, self.b+other.b, self.c+other.c, self.d+other.d)
    def __mul__(self, other):
        a1,b1,c1,d1 = self.a,self.b,self.c,self.d
        a2,b2,c2,d2 = other.a,other.b,other.c,other.d
        return Quaternion(
            a1*a2 - b1*b2 - c1*c2 - d1*d2,
            a1*b2 + b1*a2 + c1*d2 - d1*c2,
            a1*c2 - b1*d2 + c1*a2 + d1*b2,
            a1*d2 + b1*c2 - c1*b2 + d1*a2
        )
    def __str__(self): return f"{self.a:.6f} + {self.b:.6f}i + {self.c:.6f}j + {self.d:.6f}k"
    def norm(self): return math.sqrt(self.a**2 + self.b**2 + self.c**2 + self.d**2)
    def conjugate(self): return Quaternion(self.a, -self.b, -self.c, -self.d)
    def inverse(self):
        n2 = self.norm()**2
        if n2 == 0: return Quaternion(0,0,0,0)
        return Quaternion(self.a/n2, -self.b/n2, -self.c/n2, -self.d/n2)
    @staticmethod
    def from_axis_angle(axis: List[float], angle: float):
        norm = math.sqrt(axis[0]**2 + axis[1]**2 + axis[2]**2)
        if norm == 0: return Quaternion(1,0,0,0)
        x,y,z = axis[0]/norm, axis[1]/norm, axis[2]/norm
        h = angle/2
        s = math.sin(h)
        return Quaternion(math.cos(h), x*s, y*s, z*s)

def rotate_pos(pos: Tuple[float,float,float]) -> Tuple[float,float,float]:
    x, y, z = pos
    return (-y, x, z)

# ------------------------------------------------------------------------------
# 1/2 PRIME RESONATOR
# ------------------------------------------------------------------------------
class PrimeResonator:
    def __init__(self, prime: int, spectral_gap: float, pos: Tuple[float,float,float]):
        self.prime = prime
        self.spectral_gap = spectral_gap
        self.pos = pos
        self.frequency = math.log(prime) * spectral_gap          # ω_p ∝ log(p)
        axis = [1/math.sqrt(3), 1/math.sqrt(3), 1/math.sqrt(3)]
        self.state = Quaternion.from_axis_angle(axis, self.frequency)
        self.energy = math.sqrt(math.log(prime)**2 + (math.pi/2)**2) * spectral_gap

    def evolve(self, dt: float, coupling: Quaternion):
        if coupling.norm() > 0:
            h_norm = coupling.norm()
            axis = [coupling.b/h_norm, coupling.c/h_norm, coupling.d/h_norm]
            evolution = Quaternion.from_axis_angle(axis, h_norm * dt)
            self.state = evolution * self.state
            self.frequency += 0.01 * (h_norm - 1) * dt

# ------------------------------------------------------------------------------
# RESONATOR LATTICE
# ------------------------------------------------------------------------------
class ResonatorLattice:
    def __init__(self, primes: List[int], spectral_gaps: List[float]):
        self.primes = primes
        self.spec_gaps = spectral_gaps
        self.n = min(len(primes), len(spectral_gaps))
        # Golden‑ratio spiral positions
        self.positions = self._golden_positions()
        # Original and rotated resonators
        self.ress = [PrimeResonator(self.primes[i], self.spec_gaps[i%len(self.spec_gaps)], self.positions[i])
                     for i in range(self.n)]
        self.ress_rot = [PrimeResonator(self.primes[i], self.spec_gaps[i%len(self.spec_gaps)], rotate_pos(self.positions[i]))
                         for i in range(self.n)]
        # Quaternion couplings
        self.coupling = self._init_coupling(self.positions)
        self.coupling_rot = self._init_coupling([rotate_pos(p) for p in self.positions])
        # Frequency gaps
        self.freq_gaps = self._calc_freq_gaps(self.ress)
        self.freq_gaps_rot = self._calc_freq_gaps(self.ress_rot)
        # Correlations with spectral gaps
        self.corr = self._correlate(self.freq_gaps, self.spec_gaps)
        self.corr_rot = self._correlate(self.freq_gaps_rot, self.spec_gaps)

    def _golden_positions(self) -> List[Tuple[float,float,float]]:
        phi = (1 + math.sqrt(5))/2
        pos = []
        for i in range(self.n):
            x = phi * i * math.cos(2*math.pi*phi*i)
            y = phi * i * math.sin(2*math.pi*phi*i)
            z = phi * i * math.sin(2*math.pi*phi**2*i)
            scale = 10.0/(i+1)
            pos.append((x*scale, y*scale, z*scale))
        return pos

    def _init_coupling(self, positions) -> Dict[Tuple[int,int], Quaternion]:
        coup = {}
        for i in range(self.n):
            for j in range(i+1, self.n):
                xi,yi,zi = positions[i]
                xj,yj,zj = positions[j]
                dx,dy,dz = xj-xi, yj-yi, zj-zi
                dist = math.sqrt(dx*dx + dy*dy + dz*dz)
                if dist < 1e-10: continue
                strength = math.log(self.primes[j])/math.log(self.primes[i]) / dist
                coup[(i,j)] = Quaternion(0, strength*dx/dist, strength*dy/dist, strength*dz/dist)
        return coup

    def _calc_freq_gaps(self, resonators) -> List[float]:
        freqs = sorted([r.frequency for r in resonators])
        return [freqs[i+1]-freqs[i] for i in range(len(freqs)-1)]

    def _normalize(self, arr):
        if len(arr) < 2: return arr
        m = sum(arr)/len(arr)
        s = math.sqrt(sum((x-m)**2 for x in arr)/len(arr)) or 1.0
        return [(x-m)/s for x in arr]

    def _pearson(self, x, y):
        if len(x) != len(y) or len(x) < 2: return 0.0
        n = len(x)
        sx = sy = sxy = sx2 = sy2 = 0.0
        for i in range(n):
            sx += x[i]; sy += y[i]; sxy += x[i]*y[i]
            sx2 += x[i]**2; sy2 += y[i]**2
        den = math.sqrt((n*sx2 - sx**2)*(n*sy2 - sy**2))
        return (n*sxy - sx*sy)/den if den != 0 else 0.0

    def _correlate(self, lattice_gaps, spectral_gaps):
        m = min(len(lattice_gaps), len(spectral_gaps))
        if m < 2: return 0.0
        lat_n = self._normalize(lattice_gaps[:m])
        spe_n = self._normalize(spectral_gaps[:m])
        return self._pearson(lat_n, spe_n)

    def evolve(self, dt: float = 0.01, steps: int = 30):
        corrs = []
        for step in range(steps):
            for i in range(self.n):
                total = Quaternion(0,0,0,0)
                for j in range(self.n):
                    if i != j:
                        key = (min(i,j), max(i,j))
                        if key in self.coupling:
                            total = total + self.coupling[key]
                self.ress[i].evolve(dt, total)
            self.freq_gaps = self._calc_freq_gaps(self.ress)
            corrs.append(self._correlate(self.freq_gaps, self.spec_gaps))
        return corrs

    def evolve_rotated(self, dt: float = 0.01, steps: int = 30):
        corrs = []
        for step in range(steps):
            for i in range(self.n):
                total = Quaternion(0,0,0,0)
                for j in range(self.n):
                    if i != j:
                        key = (min(i,j), max(i,j))
                        if key in self.coupling_rot:
                            total = total + self.coupling_rot[key]
                self.ress_rot[i].evolve(dt, total)
            self.freq_gaps_rot = self._calc_freq_gaps(self.ress_rot)
            corrs.append(self._correlate(self.freq_gaps_rot, self.spec_gaps))
        return corrs

# QUANTUM SPECTRAL CORRELATOR
# ------------------------------------------------------------------------------
class QuantumSpectralCorrelator:
    def __init__(self, n_modes: int = 28):
        self.n_modes = n_modes
        self.primes = generate_primes(n_modes)
        self.spec_gaps = spectral_gaps(n_modes)
        self.lattice = ResonatorLattice(self.primes, self.spec_gaps)
        # Approximate eigenvalues of Hamiltonian H = Σ log(p)·a†a + Σ J_pq·Q_pq
        self.eigenvals = self._approx_eigenvals()
        self.quantum_gaps = self._calc_quantum_gaps()

    def _approx_eigenvals(self) -> List[float]:
        ev = []
        for i in range(self.n_modes):
            e = self.lattice.ress[i].energy
            coup_sum = 0.0
            for j in range(self.n_modes):
                if i != j:
                    key = (min(i,j), max(i,j))
                    if key in self.lattice.coupling:
                        coup_sum += self.lattice.coupling[key].norm()
            ev.append(e + 0.5 * coup_sum)
        ev.sort()
        return ev

    def _calc_quantum_gaps(self) -> List[float]:
        return [self.eigenvals[i+1]-self.eigenvals[i] for i in range(len(self.eigenvals)-1)]

    def analyze(self):
        print("\n" + "="*80)
        print("QUANTUM SPECTRAL CORRELATOR")
        print("="*80)
        print(f"Primes used: {self.primes[:5]} ...")
        print(f"Riemann zeta gaps (first 5): {[f'{g:.4f}' for g in self.spec_gaps[:5]]}")
        print(f"Original lattice correlation: {self.lattice.corr:.6f}")
        print(f"R₉₀ rotated lattice correlation: {self.lattice.corr_rot:.6f}")
        # Evolve both lattices
        orig_corrs = self.lattice.evolve(steps=20)
        rot_corrs = self.lattice.evolve_rotated(steps=20)
        print(f"Final correlation (original): {orig_corrs[-1]:.6f}")
        print(f"Final correlation (rotated):  {rot_corrs[-1]:.6f}")
        # Quantum gaps vs zeta gaps
        qg = self.quantum_gaps
        zg = self.spec_gaps
        m = min(len(qg), len(zg))
        qn = self.lattice._normalize(qg[:m])
        zn = self.lattice._normalize(zg[:m])
        q_zeta = self.lattice._pearson(qn, zn)
        print(f"Quantum gaps vs Riemann gaps correlation: {q_zeta:.6f}")
        chain = q_zeta * self.lattice.corr * orig_corrs[-1]
        print(f"Chain correlation product: {chain:.6f}")
        if abs(chain) > 0.3:
            print("🎯 SIGNIFICANT CORRELATION DETECTED!")
        return self.quantum_gaps

# BITCOIN PUZZLE #71 & VALKNUT DIAL v9
# ------------------------------------------------------------------------------
class ValknutPuzzleSolver:
    def __init__(self):
        self.range_start = 2**70
        self.range_end = 2**71 - 1
        self.predicted_key = 0x400000000000000003
        # Valknut parameters
        self.hbar = 1.0545718e-34
        self.gamma = 1.0
        self.m = 1.0
        self.v_eff = 1.0
        self.xi_tolerance = 0.016

    # ----- Base search point generation (from quantum gaps) -----
    def generate_search_offsets(self, quantum_gaps: List[float], n_points: int = 20) -> List[Dict]:
        points = []
        for i, gap in enumerate(quantum_gaps[:n_points]):
            offset = int(abs(gap) * 1e6) % 1000000 - 500000
            key = self.predicted_key + offset
            if self.range_start <= key <= self.range_end:
                points.append({
                    'index': i+1,
                    'gap': gap,
                    'offset': offset,
                    'key': key,
                    'hex': hex(key),
                    'prob': 1.0/(abs(offset)+1)
                })
        points.sort(key=lambda x: x['prob'], reverse=True)
        return points

    # ----- Valknut v9 components -----
    def _zeta_like(self, data: bytes) -> float:
        h = hashlib.sha256(data).hexdigest()
        return int(h[:8], 16) / (2**32)

    def _thue_morse_phase(self, data: bytes) -> float:
        seq = [bin(b).count('1') % 2 for b in data[:257]]
        cum = 0
        for bit in seq:
            cum = (cum + bit) % 2
        return (cum * 2 * math.pi) % (2*math.pi) / (2*math.pi)

    def _geom_curv_at_fib(self, data: bytes, fib: int) -> float:
        h = hashlib.sha256(data + str(fib).encode()).hexdigest()
        return int(h[:4], 16) / 65535

    # ----- FUSED DYSON FACTOR (Relativistic Equilibrium) -----
    def _dyson_factor(self, data: bytes) -> float:
        """
        Compute dimensionless Dyson equilibrium factor from block data.
        Uses the exact relativistic pressure formula for a thin shell.
        Returns a value in [0,2] centered around 1 for perfect equilibrium.
        """
        h = hashlib.sha256(data).hexdigest()
        # Extract physical parameters (scaled to reasonable ranges)
        M_sun = int(h[:4], 16) / 65535 * 10          # 0..10 solar masses
        R_sun = int(h[4:8], 16) / 65535 * 20         # 0..20 solar radii
        rho_gcm3 = int(h[8:12], 16) / 65535 * 100    # 0..100 g/cm³

        # Convert to SI units
        M_kg = M_sun * 1.989e30
        R_m = R_sun * 6.957e8
        rho_kgm3 = rho_gcm3 * 1000.0

        # Constants
        G = 6.67430e-11
        c = 299792458.0
        # Critical density (example: water density, could be tuned)
        rho_c = 1000.0  # kg/m³

        # Compute denominator with safeguard
        denom = 2.0 * (R_m - 2.0 * G * M_kg) * (1.0 + rho_kgm3 / rho_c)
        if denom <= 0 or abs(denom) < 1e-30:
            # If the shell is inside the Schwarzschild radius or unphysical, return extreme
            return 2.0  # max tension

        P_t = (G * M_kg * rho_kgm3 * R_m) / denom

        # Reference pressure for normalization (e.g., 1e10 Pa, arbitrary)
        P_ref = 1e10
        # Use tanh to map [0,∞) to [0,1) and then scale to [0,2]
        dyson = math.tanh(P_t / P_ref) * 2.0
        return max(0.0, min(2.0, dyson))

    def xi_valknut_v9(self, data: bytes) -> float:
        # 1. spectral cube
        spec_cube = self._zeta_like(data) ** 3
        # 2. Berry flow (using Thue‑Morse phase)
        berry = self._thue_morse_phase(data)
        # 3. quantum‑Fibonacci gradient
        fibs = [1,2,3,5,8,13,21]
        geom = sum(self._geom_curv_at_fib(data, f) for f in fibs)
        q_fib = (self.hbar / (self.gamma * self.m * self.v_eff)) * geom
        q_fib = max(0.0, min(2.0, q_fib))
        # 4. Dyson factor
        dyson = self._dyson_factor(data)
        xi = (spec_cube + berry + q_fib + dyson) / 4.0
        return xi

    # ----- WIF generation (Base58) -----
    def _base58_encode(self, data: bytes) -> str:
        alphabet = '123456789ABCDEFGHJKLMNPQRSTUVWXYZabcdefghijkmnopqrstuvwxyz'
        n = int.from_bytes(data, 'big')
        result = ''
        while n > 0:
            n, rem = divmod(n, 58)
            result = alphabet[rem] + result
        zeros = 0
        for b in data:
            if b == 0:
                zeros += 1
            else:
                break
        return '1'*zeros + result

    def wif_from_key(self, key: int) -> str:
        prefix = b'\x80'
        key_bytes = key.to_bytes(32, 'big')
        suffix = b'\x01'
        data = prefix + key_bytes + suffix
        h1 = hashlib.sha256(data).digest()
        h2 = hashlib.sha256(h1).digest()
        checksum = h2[:4]
        return self._base58_encode(data + checksum)

    # ----- Valknut‑filtered candidates -----
    def filtered_candidates(self, quantum_gaps: List[float], n_raw: int = 30) -> List[Dict]:
        raw = self.generate_search_offsets(quantum_gaps, n_raw)
        filtered = []
        for pt in raw:
            key_bytes = pt['key'].to_bytes(32, 'big')
            xi = self.xi_valknut_v9(key_bytes)
            if abs(xi - 1.0) <= self.xi_tolerance:
                pt['xi'] = xi
                pt['wif'] = self.wif_from_key(pt['key'])
                filtered.append(pt)
        filtered.sort(key=lambda x: abs(x['xi'] - 1.0))
        return filtered

    def analyze(self, quantum_gaps: List[float]):
        print("\n" + "="*80)
        print("VALKNUT v9 DIAL FILTERING")
        print("="*80)
        candidates = self.filtered_candidates(quantum_gaps, 30)

            print(f"     Offset: {p['offset']:+} (gap {p['gap']:.6f})")
            print(f"     Valknut ξ = {p['xi']:.6f}")
            print(f"     WIF: {p['wif']}")
        return candidates
